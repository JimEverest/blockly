GetCPParams


KeyWords:

portName: string

targetAcc: float

junctionVel: float

isRealTimeTrack: bool

*acc: float(isRealTimeTrack = false)

*period: float(isRealTimeTrack = true)

 

INPUT:

{
    "id": 1,
    "jsonrpc": "2.0",
    "method": "dobotlink.MagicianLite.GetCPParams",
    "params": {
        "portName": "COM4"
    }
}

 

OUTPUT:

{
    "id": 1,
    "jsonrpc": "2.0",
    "result": {
        "targetAcc": 200,
        "junctionVel": 100,
        "isRealTimeTrack": false,
        "acc": 100
    }
}
